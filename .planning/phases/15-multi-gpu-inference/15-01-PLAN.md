```cpp
/** Multi-GPU state managed by wan_context */
struct MultiGPUState {
    std::vector<ggml_backend_t> backends;           // One per GPU
    ggml_backend_sched_t sched = nullptr;           // Multi-backend scheduler
    std::vector<float> tensor_split;                // Split ratios per GPU
    int main_device = 0;                            // Primary GPU device ID
    int num_devices = 0;                            // Number of active GPUs
    wan_distribution_strategy_t strategy = WAN_DIST_AUTO;
    bool initialized = false;

    ~MultiGPUState() {
        if (sched) {
            ggml_backend_sched_free(sched);
            sched = nullptr;
        }
        for (auto& b : backends) {
            if (b) ggml_backend_free(b);
        }
        backends.clear();
    }
};
```

```cpp
WAN_API wan_error_t wan_params_set_gpu_ids(wan_params_t* params, const int* gpu_ids, int num_gpus) {
    if (!params || !gpu_ids || num_gpus <= 0 || num_gpus > WAN_MAX_GPUS) {
        return WAN_ERROR_INVALID_ARGUMENT;
    }
    memcpy(params->gpu_ids, gpu_ids, num_gpus * sizeof(int));
    params->num_gpus = num_gpus;
    return WAN_SUCCESS;
}

WAN_API wan_error_t wan_params_set_num_gpus(wan_params_t* params, int num_gpus) {
    if (!params || num_gpus < 0 || num_gpus > WAN_MAX_GPUS) {
        return WAN_ERROR_INVALID_ARGUMENT;
    }
    params->num_gpus = num_gpus;
    for (int i = 0; i < num_gpus; i++) {
        params->gpu_ids[i] = i;
    }
    return WAN_SUCCESS;
}

WAN_API void wan_params_set_distribution_strategy(wan_params_t* params, wan_distribution_strategy_t strategy) {
    if (params) {
        params->distribution_strategy = strategy;
    }
}
```

2. Add NCCL detection and linking inside the `if(WAN_CUDA)` block, AFTER the CUDA_GRAPHS section:
```cmake
    if(WAN_NCCL)
        find_package(PkgConfig QUIET)
        if(PkgConfig_FOUND)
            pkg_check_modules(NCCL QUIET nccl)
        endif()
        if(NOT NCCL_FOUND)
            find_library(NCCL_LIBRARIES NAMES nccl HINTS /usr/lib/x86_64-linux-gnu /usr/local/lib)
            find_path(NCCL_INCLUDE_DIRS NAMES nccl.h HINTS /usr/include /usr/local/include)
            if(NCCL_LIBRARIES AND NCCL_INCLUDE_DIRS)
                set(NCCL_FOUND TRUE)
            endif()
        endif()
        if(NCCL_FOUND)
            add_definitions(-DWAN_USE_NCCL)
            message("-- NCCL found: ${NCCL_LIBRARIES}")
        else()
            message(WARNING "-- NCCL not found, multi-GPU communication will use GGML internal transfers")
            set(WAN_NCCL OFF)
        endif()
    endif()
```

3. After the library target is created (after `add_library(${WAN_LIB} ...)`), add NCCL linking:
```cmake
if(WAN_NCCL AND NCCL_FOUND)
    target_include_directories(${WAN_LIB} PRIVATE ${NCCL_INCLUDE_DIRS})
    target_link_libraries(${WAN_LIB} PRIVATE ${NCCL_LIBRARIES})
endif()
```
