```cpp
WAN_API wan_error_t wan_get_gpu_info(int* device_count,
                                      char descriptions[][256],
                                      int max_devices) {
#ifdef WAN_USE_MULTI_GPU
    int count = ggml_backend_cuda_get_device_count();
    if (device_count) *device_count = count;
    if (descriptions && max_devices > 0) {
        int n = count < max_devices ? count : max_devices;
        for (int i = 0; i < n; i++) {
            ggml_backend_cuda_get_device_description(i, descriptions[i], 256);
        }
    }
    return WAN_SUCCESS;
#else
    if (device_count) *device_count = 0;
    return WAN_ERROR_UNSUPPORTED_OPERATION;
#endif
}
```
