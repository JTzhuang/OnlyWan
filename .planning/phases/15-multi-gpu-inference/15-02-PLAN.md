```cpp
#ifdef WAN_USE_MULTI_GPU
#include "ggml-cuda.h"

static std::unique_ptr<MultiGPUState> create_multi_gpu_state(
    const int* gpu_ids, int num_gpus,
    wan_distribution_strategy_t strategy)
{
    auto state = std::make_unique<MultiGPUState>();

    // Validate device count
    int available = ggml_backend_cuda_get_device_count();
    if (available < num_gpus) {
        LOG_ERROR("Requested %d GPUs but only %d available", num_gpus, available);
        return nullptr;
    }

    // Validate GPU homogeneity (user decision: same GPU type only)
    char desc0[256];
    ggml_backend_cuda_get_device_description(gpu_ids[0], desc0, sizeof(desc0));
    for (int i = 1; i < num_gpus; i++) {
        char desc[256];
        ggml_backend_cuda_get_device_description(gpu_ids[i], desc, sizeof(desc));
        if (strcmp(desc0, desc) != 0) {
            LOG_ERROR("Mixed GPU types not supported: GPU %d is '%s', GPU %d is '%s'",
                      gpu_ids[0], desc0, gpu_ids[i], desc);
            return nullptr;
        }
    }

    // Initialize backends for each GPU
    for (int i = 0; i < num_gpus; i++) {
        ggml_backend_t backend = ggml_backend_cuda_init(gpu_ids[i]);
        if (!backend) {
            LOG_ERROR("Failed to initialize CUDA backend for device %d", gpu_ids[i]);
            return nullptr;
        }
        state->backends.push_back(backend);
    }

    // Equal tensor split ratios
    state->tensor_split.resize(num_gpus, 1.0f / num_gpus);
    state->main_device = gpu_ids[0];
    state->num_devices = num_gpus;
    state->strategy = strategy;

    // Create backend scheduler
    state->sched = ggml_backend_sched_new(
        state->backends.data(),
        nullptr,  // auto-detect buffer types
        num_gpus,
        GGML_DEFAULT_GRAPH_SIZE,
        false,    // sequential execution
        true      // op offload
    );
    if (!state->sched) {
        LOG_ERROR("Failed to create multi-GPU backend scheduler");
        return nullptr;
    }

    state->initialized = true;
    LOG_INFO("Multi-GPU initialized: %d GPUs, strategy=%d, main_device=%d",
             num_gpus, (int)strategy, gpu_ids[0]);
    return state;
}
#endif // WAN_USE_MULTI_GPU
```

    /** Allocate params buffer using split buffer type for tensor parallelism.
     *  Distributes weight tensors across multiple CUDA devices.
     *  @param main_device Primary GPU device ID
     *  @param tensor_split Array of split ratios per device (must sum to ~1.0)
     *  @param num_devices Number of devices
     */
    bool alloc_params_buffer_split(int main_device, const float* tensor_split, int num_devices) {
#ifdef WAN_USE_MULTI_GPU
        size_t num_tensors = ggml_tensor_num(params_ctx);
        ggml_backend_buffer_type_t split_buft =
            ggml_backend_cuda_split_buffer_type(main_device, tensor_split);
        if (!split_buft) {
            LOG_ERROR("%s failed to get split buffer type for device %d",
                      get_desc().c_str(), main_device);
            return false;
        }
        params_buffer = ggml_backend_alloc_ctx_tensors_from_buft(params_ctx, split_buft);
        if (params_buffer == nullptr) {
            LOG_ERROR("%s alloc split params buffer failed, num_tensors = %i",
                      get_desc().c_str(), num_tensors);
            return false;
        }
        size_t params_buffer_size = ggml_backend_buffer_get_size(params_buffer);
        LOG_DEBUG("%s split params buffer size = % 6.2f MB (across %d GPUs, %i tensors)",
                  get_desc().c_str(),
                  params_buffer_size / (1024.f * 1024.f),
                  num_devices, num_tensors);
        return true;
#else
        (void)main_device; (void)tensor_split; (void)num_devices;
        LOG_ERROR("alloc_params_buffer_split requires WAN_USE_MULTI_GPU");
        return false;
#endif
    }

1. In `src/ggml_extend.hpp`, add a new method to `GGMLRunner` that accepts an external scheduler for graph execution. Add this method AFTER the existing `compute()` method:

```cpp
    /** Compute using external backend scheduler (for multi-GPU).
     *  The scheduler handles tensor placement and cross-device communication.
     *  @param get_graph Callback returning the compute graph (type: std::function<struct ggml_cgraph*()>)
     *  @param n_threads CPU thread count for CPU backend ops
     *  @param sched The multi-backend scheduler from MultiGPUState
     *  @param output Optional output tensor pointer
     *  @param output_ctx Optional context for allocating output tensor
     */
    bool compute_with_sched(get_graph_cb_t get_graph,
                            int n_threads,
                            ggml_backend_sched_t sched,
                            struct ggml_tensor** output = nullptr,
                            struct ggml_context* output_ctx = nullptr) {
        if (!sched) return false;

        reset_compute_ctx();
        struct ggml_cgraph* gf = get_compute_graph(get_graph);

        // Use scheduler to allocate and compute the graph across backends
        ggml_backend_sched_reset(sched);
        if (!ggml_backend_sched_alloc_graph(sched, gf)) {
            LOG_ERROR("%s: failed to allocate graph on multi-GPU scheduler", get_desc().c_str());
            return false;
        }

        // Copy input tensor data to backend
        copy_data_to_backend_tensor();

        ggml_status status = ggml_backend_sched_graph_compute(sched, gf);
        if (status != GGML_STATUS_SUCCESS) {
            LOG_ERROR("%s: multi-GPU compute failed: %s", get_desc().c_str(), ggml_status_to_string(status));
            return false;
        }

        copy_cache_tensors_to_cache_buffer();
        if (output != nullptr) {
            auto result = ggml_get_tensor(compute_ctx, final_result_name.c_str());
            if (*output == nullptr && output_ctx != nullptr) {
                *output = ggml_dup_tensor(output_ctx, result);
            }
            if (*output != nullptr) {
                // Get result from whichever backend computed it
                ggml_backend_t result_backend = ggml_backend_sched_get_tensor_backend(sched, result);
                if (result_backend) {
                    ggml_ext_backend_tensor_get_and_sync(result_backend, result, (*output)->data, 0, ggml_nbytes(*output));
                }
            }
        }
        return true;
    }
```
