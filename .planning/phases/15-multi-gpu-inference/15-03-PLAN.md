/** Generate multiple videos in parallel across GPUs (data parallelism)
 *
 * Each request runs on a separate GPU with an independent model copy.
 * Requires CUDA backend and multiple GPUs.
 *
 * @param model_path Path to the model file
 * @param prompts Array of text prompts
 * @param output_paths Array of output file paths
 * @param params_array Array of generation parameters (one per request)
 * @param num_requests Number of generation requests
 * @param gpu_ids Array of GPU device IDs to use
 * @param num_gpus Number of GPUs available
 * @param results Output array of error codes (one per request, caller allocates)
 * @return WAN_SUCCESS if all requests completed (check individual results for per-request status)
 */
wan_error_t wan_generate_batch_t2v(const char* model_path,
                                    const char** prompts,
                                    const char** output_paths,
                                    const wan_params_t** params_array,
                                    int num_requests,

2. In `wan-api.cpp`, implement `wan_generate_batch_t2v`. Add `#include <thread>` at the top includes. The implementation:

```cpp
WAN_API wan_error_t wan_generate_batch_t2v(const char* model_path,
                                            const char** prompts,
                                            const char** output_paths,
                                            const wan_params_t** params_array,
                                            int num_requests,
                                            const int* gpu_ids,
                                            int num_gpus,
                                            wan_error_t* results) {
    if (!model_path || !prompts || !output_paths || !params_array ||
        !results || num_requests <= 0 || !gpu_ids || num_gpus <= 0) {
        return WAN_ERROR_INVALID_ARGUMENT;
    }

#ifdef WAN_USE_MULTI_GPU
    // Initialize: zero out results
    for (int i = 0; i < num_requests; i++) {
        results[i] = WAN_ERROR_UNKNOWN;
    }

    // Worker function: load model on specific GPU, generate video
    auto worker = [&](int request_idx, int gpu_id) {
        // Create single-GPU params targeting specific device
        wan_context_t* ctx = nullptr;
        char gpu_backend[32];
        snprintf(gpu_backend, sizeof(gpu_backend), "cuda");

        wan_error_t err = wan_load_model(model_path,
                                          params_array[request_idx]->n_threads,
                                          gpu_backend, &ctx);
        if (err != WAN_SUCCESS) {
            results[request_idx] = err;
            return;
        }

        err = wan_generate_video_t2v_ex(ctx, prompts[request_idx],
                                         params_array[request_idx],
                                         output_paths[request_idx]);
        results[request_idx] = err;
        wan_free(ctx);
    };

    // Distribute requests across GPUs using round-robin
    std::vector<std::thread> threads;
    for (int i = 0; i < num_requests; i++) {
        int gpu_id = gpu_ids[i % num_gpus];
        threads.emplace_back(worker, i, gpu_id);
    }

    // Wait for all threads
    for (auto& t : threads) {
        if (t.joinable()) t.join();
    }

    // Check if any request failed
    for (int i = 0; i < num_requests; i++) {
        if (results[i] != WAN_SUCCESS) {
            return WAN_ERROR_GENERATION_FAILED;  // At least one failed
        }
    }
    return WAN_SUCCESS;
#else
    return WAN_ERROR_UNSUPPORTED_OPERATION;
#endif
}
```

Note: This is a straightforward data-parallel implementation where each GPU loads its own model copy and processes requests independently. The `wan_load_model` currently always uses device 0 -- the executor should modify the worker to set `CUDA_VISIBLE_DEVICES` or use the gpu_id parameter to target specific devices. The simplest approach is to modify `WanBackend::create` to accept a device_id parameter (already done in Plan 02's `create_multi_gpu_state`), or use the existing `ggml_backend_cuda_init(gpu_id)` directly.

For a cleaner implementation, the executor may choose to:
- Add a `wan_load_model_on_device(model_path, n_threads, device_id, &ctx)` internal helper
- Or pass device_id through params->gpu_ids[0] and have wan_load_model read it

The key constraint is that each thread must initialize its own CUDA context on a different device.
