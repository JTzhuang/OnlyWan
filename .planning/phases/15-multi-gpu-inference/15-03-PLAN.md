1. In `wan_loader.cpp`, add a `wan_load_model_on_device` internal helper that loads a model targeting a specific CUDA device. This reuses the `ggml_backend_cuda_init(device_id)` pattern established in Plan 02's `create_multi_gpu_state`:

```cpp
#ifdef WAN_USE_MULTI_GPU
/** Load model targeting a specific CUDA device for data-parallel execution.
 *  Creates a single-GPU context pinned to the given device_id.
 */
static wan_error_t wan_load_model_on_device(const char* model_path,
                                             int n_threads,
                                             int device_id,
                                             wan_context_t** out_ctx) {
    if (!model_path || !out_ctx) return WAN_ERROR_INVALID_ARGUMENT;

    // Validate device_id
    int available = ggml_backend_cuda_get_device_count();
    if (device_id < 0 || device_id >= available) {
        LOG_ERROR("Invalid device_id %d (available: %d)", device_id, available);
        return WAN_ERROR_INVALID_ARGUMENT;
    }

    std::unique_ptr<wan_context> ctx(new wan_context());
    ctx->model_path = model_path;
    ctx->n_threads = n_threads;
    ctx->backend_type = "cuda";

    // Load model weights to CPU
    WanModelLoadResult result = Wan::WanModel::load(ctx->model_path);
    if (!result.success) {
        set_last_error(ctx.get(), result.error_message.c_str());
        return WAN_ERROR_MODEL_LOAD_FAILED;
    }

    ctx->wan_runner  = result.wan_runner;
    ctx->vae_runner  = result.vae_runner;
    ctx->t5_embedder = result.t5_embedder;
    ctx->clip_runner = result.clip_runner;
    ctx->model_type  = result.model_type;
    ctx->params.model_version = result.model_version;

    // Create backend targeting specific device
    ggml_backend_t cuda_backend = ggml_backend_cuda_init(device_id);
    if (!cuda_backend) {
        set_last_error(ctx.get(), "Failed to init CUDA backend on device");
        return WAN_ERROR_BACKEND_FAILED;
    }

    // Wrap in WanBackend (executor: check WanBackend constructor or use
    // WanBackend::create then override the internal backend pointer)
    ctx->backend = WanBackendPtr(Wan::WanBackend::create("cuda", n_threads));
    if (!ctx->backend) {
        ggml_backend_free(cuda_backend);
        return WAN_ERROR_BACKEND_FAILED;
    }

    // Auto-enable flash attention
    if (ctx->wan_runner) ctx->wan_runner->set_flash_attention_enabled(true);
    if (ctx->vae_runner) ctx->vae_runner->set_flash_attention_enabled(true);
    if (ctx->clip_runner) ctx->clip_runner->set_flash_attention_enabled(true);

    *out_ctx = ctx.release();
    return WAN_SUCCESS;
}
#endif
```

Note: The executor should verify how `WanBackend::create` selects the CUDA device internally. If it always uses device 0, the executor must either:
- Add a `device_id` parameter to `WanBackend::create`, or
- Set `CUDA_VISIBLE_DEVICES` per-thread (less clean), or
- Directly construct the backend and assign it to the runners' `runtime_backend`

The cleanest approach is adding `WanBackend::create(const std::string& type, int n_threads, int device_id = 0)`.

3. In `wan-api.cpp`, implement `wan_generate_batch_t2v`. Add `#include <thread>` at the top includes. The implementation uses `wan_load_model_on_device` to pin each worker to a specific GPU:

```cpp
WAN_API wan_error_t wan_generate_batch_t2v(const char* model_path,
                                            const char** prompts,
                                            const char** output_paths,
                                            const wan_params_t** params_array,
                                            int num_requests,
                                            const int* gpu_ids,
                                            int num_gpus,
                                            wan_error_t* results) {
    if (!model_path || !prompts || !output_paths || !params_array ||
        !results || num_requests <= 0 || !gpu_ids || num_gpus <= 0) {
        return WAN_ERROR_INVALID_ARGUMENT;
    }

#ifdef WAN_USE_MULTI_GPU
    // Initialize: zero out results
    for (int i = 0; i < num_requests; i++) {
        results[i] = WAN_ERROR_UNKNOWN;
    }

    // Worker function: load model on specific GPU, generate video
    auto worker = [&](int request_idx, int gpu_id) {
        wan_context_t* ctx = nullptr;

        // Load model targeting specific GPU device
        wan_error_t err = wan_load_model_on_device(
            model_path, params_array[request_idx]->n_threads,
            gpu_id, &ctx);
        if (err != WAN_SUCCESS) {
            results[request_idx] = err;
            return;
        }

        err = wan_generate_video_t2v_ex(ctx, prompts[request_idx],
                                         params_array[request_idx],
                                         output_paths[request_idx]);
        results[request_idx] = err;
        wan_free(ctx);
    };

    // Distribute requests across GPUs using round-robin
    std::vector<std::thread> threads;
    for (int i = 0; i < num_requests; i++) {
        int gpu_id = gpu_ids[i % num_gpus];
        threads.emplace_back(worker, i, gpu_id);
    }

    // Wait for all threads
    for (auto& t : threads) {
        if (t.joinable()) t.join();
    }

    // Check if any request failed
    for (int i = 0; i < num_requests; i++) {
        if (results[i] != WAN_SUCCESS) {
            return WAN_ERROR_GENERATION_FAILED;  // At least one failed
        }
    }
    return WAN_SUCCESS;
#else
    return WAN_ERROR_UNSUPPORTED_OPERATION;
#endif
}
```
