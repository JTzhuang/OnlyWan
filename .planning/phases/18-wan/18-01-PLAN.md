## Architecture: Per-Type Singleton Registries (Approach C)

Three separate singleton registries — simplest correct design with zero template complexity:

```
src/model_registry.hpp
├── PerTypeRegistry<ModelType> template (fully inline via Meyers singleton)
│   ├── register_factory(version_string, factory_fn)
│   ├── create(version_string, backend, offload, tensor_map, prefix) -> unique_ptr<ModelType>
│   ├── has_version(version_string) -> bool
│   └── static instance() -> PerTypeRegistry& (magic static, init-once)
├── using CLIPRegistry   = PerTypeRegistry<CLIPTextModelRunner>
├── using T5Registry     = PerTypeRegistry<T5Runner>
├── using WanVAERegistry = PerTypeRegistry<WAN::WanVAERunner>
├── REGISTER_CLIP_FACTORY(version_string, factory_lambda)
├── REGISTER_T5_FACTORY(version_string, factory_lambda)
└── REGISTER_WAN_VAE_FACTORY(version_string, factory_lambda)

src/model_factory.cpp
└── 9 REGISTER_*_FACTORY macro invocations (static init — runs before main)
```

**model_registry.hpp structure:**
```cpp
#pragma once
#include <functional>
#include <map>
#include <memory>
#include <mutex>
#include <stdexcept>
#include <string>
#include "model.h"        // String2TensorStorage
#include "ggml-backend.h" // ggml_backend_t

// Forward declarations — full headers included only in model_factory.hpp
struct CLIPTextModelRunner;
struct T5Runner;
namespace WAN { struct WanVAERunner; }

// PerTypeRegistry<ModelType> — one singleton per model type
template<typename ModelType>
class PerTypeRegistry {
public:
    using FactoryFn = std::function<std::unique_ptr<ModelType>(
        ggml_backend_t backend,
        bool offload_params_to_cpu,
        const String2TensorStorage& tensor_map,
        const std::string& prefix
    )>;

    static PerTypeRegistry& instance() {
        static PerTypeRegistry registry;  // C++11 magic static — init-once, thread-safe
        return registry;
    }

    void register_factory(const std::string& version, FactoryFn fn) {
        std::lock_guard<std::mutex> lock(mutex_);
        factories_[version] = std::move(fn);
    }

    std::unique_ptr<ModelType> create(
            const std::string& version,
            ggml_backend_t backend,
            bool offload,
            const String2TensorStorage& map,
            const std::string& prefix = "") const {
        std::lock_guard<std::mutex> lock(mutex_);
        auto it = factories_.find(version);
        if (it == factories_.end()) {
            throw std::runtime_error("Unknown model version: " + version);
        }
        return it->second(backend, offload, map, prefix);
    }

    bool has_version(const std::string& version) const {
        std::lock_guard<std::mutex> lock(mutex_);
        return factories_.count(version) > 0;
    }

private:
    PerTypeRegistry() = default;
    mutable std::mutex mutex_;
    std::map<std::string, FactoryFn> factories_;
};

// Per-type registry aliases
using CLIPRegistry    = PerTypeRegistry<CLIPTextModelRunner>;
using T5Registry      = PerTypeRegistry<T5Runner>;
using WanVAERegistry  = PerTypeRegistry<WAN::WanVAERunner>;

// Registration macros — anonymous-namespace struct calls register_factory() in constructor
// The struct is a static local at TU scope, so it runs before main().
#define REGISTER_CLIP_FACTORY(version_string, ...)                                    \
    namespace {                                                                       \
    struct _CLIPRegistrar_##__LINE__ {                                                \
        _CLIPRegistrar_##__LINE__() {                                                 \
            CLIPRegistry::instance().register_factory(version_string, __VA_ARGS__);   \
        }                                                                             \
    } _clip_reg_inst_##__LINE__;                                                      \
    }

#define REGISTER_T5_FACTORY(version_string, ...)                                      \
    namespace {                                                                       \
    struct _T5Registrar_##__LINE__ {                                                  \
        _T5Registrar_##__LINE__() {                                                   \
            T5Registry::instance().register_factory(version_string, __VA_ARGS__);     \
        }                                                                             \
    } _t5_reg_inst_##__LINE__;                                                        \
    }

#define REGISTER_WAN_VAE_FACTORY(version_string, ...)                                      \
    namespace {                                                                            \
    struct _WanVAERegistrar_##__LINE__ {                                                   \
        _WanVAERegistrar_##__LINE__() {                                                    \
            WanVAERegistry::instance().register_factory(version_string, __VA_ARGS__);      \
        }                                                                                  \
    } _wan_vae_reg_inst_##__LINE__;                                                        \
    }
```
