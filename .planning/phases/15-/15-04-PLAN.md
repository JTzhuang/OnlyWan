The script invokes wan-cli with different GPU configurations and parses output/nvidia-smi for metrics.

```python
#!/usr/bin/env python3
"""Multi-GPU performance benchmark for wan-cpp (MGPU-10).

Captures:
  - step_latency (ms/step): Euler denoising step execution time
  - memory_usage (GB): Per-GPU VRAM consumption
  - comm_bandwidth (%): GPU interconnect utilization estimate

Usage:
  python3 scripts/benchmark_multi_gpu.py --cli ./build_cuda/bin/wan-cli \
      --model MODEL_PATH --num-gpus 2 --steps 5
"""
import argparse
import subprocess
import time
import json
import os
import sys
import re

def get_gpu_memory_usage():
    """Query per-GPU memory usage via nvidia-smi (GB)."""
    try:
        result = subprocess.run(
            ["nvidia-smi", "--query-gpu=index,memory.used,memory.total",
             "--format=csv,noheader,nounits"],
            capture_output=True, text=True, timeout=10
        )
        memory = {}
        for line in result.stdout.strip().split("\n"):
            parts = [p.strip() for p in line.split(",")]
            if len(parts) >= 3:
                idx = int(parts[0])
                used_mb = float(parts[1])
                total_mb = float(parts[2])
                memory[idx] = {
                    "used_gb": round(used_mb / 1024, 2),
                    "total_gb": round(total_mb / 1024, 2),
                    "utilization_pct": round(used_mb / total_mb * 100, 1)
                }
        return memory
    except Exception as e:
        print(f"Warning: nvidia-smi query failed: {e}", file=sys.stderr)
        return {}

def get_nvlink_bandwidth():
    """Estimate communication bandwidth via nvidia-smi nvlink counters."""
    try:
        result = subprocess.run(
            ["nvidia-smi", "nvlink", "--status"],
            capture_output=True, text=True, timeout=10
        )
        if result.returncode == 0 and "NVLink" in result.stdout:
            links = result.stdout.count("active")
            return {"nvlink_active_links": links, "comm_available": True}
    except Exception:
        pass
    return {"nvlink_active_links": 0, "comm_available": False}

def run_benchmark(cli_path, model_path, num_gpus, steps, prompt, output_dir):
    """Run wan-cli and measure step latency."""
    output_path = os.path.join(output_dir, f"bench_{num_gpus}gpu.avi")

    mem_before = get_gpu_memory_usage()

    cmd = [cli_path, "-m", model_path, "-p", prompt, "-o", output_path,
           "--steps", str(steps), "--seed", "42", "--verbose"]
    if num_gpus > 1:
        cmd.extend(["--num-gpus", str(num_gpus)])

    start = time.perf_counter()
    result = subprocess.run(cmd, capture_output=True, text=True, timeout=600)
    elapsed = time.perf_counter() - start

    mem_after = get_gpu_memory_usage()

    step_times = []
    for line in result.stdout.split("\n"):
        m = re.search(r"step\s+\d+.*?(\d+\.?\d*)\s*ms", line, re.IGNORECASE)
        if m:
            step_times.append(float(m.group(1)))

    step_latency_ms = (elapsed * 1000) / steps if steps > 0 else 0
    if step_times:
        step_latency_ms = sum(step_times) / len(step_times)

    return {
        "num_gpus": num_gpus,
        "steps": steps,
        "total_time_s": round(elapsed, 2),
        "step_latency_ms": round(step_latency_ms, 1),
        "memory_usage": mem_after,
        "exit_code": result.returncode,
        "success": result.returncode == 0
    }

def main():
    parser = argparse.ArgumentParser(description="Multi-GPU benchmark")
    parser.add_argument("--cli", required=True, help="Path to wan-cli binary")
    parser.add_argument("--model", required=True, help="Path to model file")
    parser.add_argument("--num-gpus", type=int, default=2, help="Number of GPUs")
    parser.add_argument("--steps", type=int, default=5, help="Denoising steps")
    parser.add_argument("--prompt", default="a cat walking", help="Test prompt")
    parser.add_argument("--output-dir", default="/tmp/wan_bench", help="Output dir")
    parser.add_argument("--json", action="store_true", help="Output JSON format")
    args = parser.parse_args()

    os.makedirs(args.output_dir, exist_ok=True)
    results = {}

    print("Running single-GPU baseline...")
    results["single_gpu"] = run_benchmark(
        args.cli, args.model, 1, args.steps, args.prompt, args.output_dir)

    print(f"Running {args.num_gpus}-GPU benchmark...")
    results["multi_gpu"] = run_benchmark(
        args.cli, args.model, args.num_gpus, args.steps, args.prompt, args.output_dir)

    results["comm_info"] = get_nvlink_bandwidth()

    if results["single_gpu"]["success"] and results["multi_gpu"]["success"]:
        speedup = results["single_gpu"]["step_latency_ms"] / max(results["multi_gpu"]["step_latency_ms"], 0.1)
        results["speedup"] = round(speedup, 2)
    else:
        results["speedup"] = None

    if args.json:
        print(json.dumps(results, indent=2))
    else:
        print("\n=== Benchmark Results ===")
        print(f"Single GPU step latency: {results['single_gpu']['step_latency_ms']} ms/step")
        print(f"Multi GPU step latency:  {results['multi_gpu']['step_latency_ms']} ms/step")
        if results["speedup"]:
            print(f"Speedup: {results['speedup']}x")
        print(f"\nMemory usage (multi-GPU):")
        for gpu_id, mem in results["multi_gpu"]["memory_usage"].items():
            print(f"  GPU {gpu_id}: {mem['used_gb']}/{mem['total_gb']} GB ({mem['utilization_pct']}%)")
        print(f"\nComm: NVLink links={results['comm_info']['nvlink_active_links']}")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""Numerical precision validation for multi-GPU inference (MGPU-09).

Compares single-GPU vs multi-GPU output frame-by-frame.
Threshold: < 0.01% pixel deviation.

Usage:
  python3 scripts/validate_precision.py --cli ./build_cuda/bin/wan-cli \
      --model MODEL_PATH --num-gpus 2 --steps 3
"""
import argparse
import subprocess
import os
import sys
import tempfile

def run_inference(cli_path, model_path, num_gpus, steps, seed, output_path):
    """Run wan-cli and return success status."""
    cmd = [cli_path, "-m", model_path, "-p", "a cat",
           "-o", output_path, "--steps", str(steps), "--seed", str(seed)]
    if num_gpus > 1:
        cmd.extend(["--num-gpus", str(num_gpus)])
    result = subprocess.run(cmd, capture_output=True, text=True, timeout=600)
    return result.returncode == 0, result.stderr

def compare_outputs(file_a, file_b):
    """Compare two output files byte-by-byte.
    Returns (identical, deviation_pct, diff_bytes, total_bytes).
    """
    with open(file_a, "rb") as fa, open(file_b, "rb") as fb:
        data_a = fa.read()
        data_b = fb.read()

    if len(data_a) != len(data_b):
        return False, 100.0, abs(len(data_a) - len(data_b)), max(len(data_a), len(data_b))

    if data_a == data_b:
        return True, 0.0, 0, len(data_a)

    total = len(data_a)
    diff_bytes = sum(1 for a, b in zip(data_a, data_b) if a != b)
    deviation_pct = (diff_bytes / total) * 100
    return False, deviation_pct, diff_bytes, total

def main():
    parser = argparse.ArgumentParser(description="Precision validation")
    parser.add_argument("--cli", required=True, help="Path to wan-cli")
    parser.add_argument("--model", required=True, help="Path to model")
    parser.add_argument("--num-gpus", type=int, default=2)
    parser.add_argument("--steps", type=int, default=3)
    parser.add_argument("--seed", type=int, default=42)
    parser.add_argument("--threshold", type=float, default=0.01,
                        help="Max deviation %% (default 0.01)")
    args = parser.parse_args()

    with tempfile.TemporaryDirectory() as tmpdir:
        single_out = os.path.join(tmpdir, "single_gpu.avi")
        multi_out = os.path.join(tmpdir, "multi_gpu.avi")

        print(f"Running single-GPU inference (seed={args.seed}, steps={args.steps})...")
        ok, err = run_inference(args.cli, args.model, 1, args.steps, args.seed, single_out)
        if not ok:
            print(f"FAIL: Single-GPU inference failed: {err}", file=sys.stderr)
            sys.exit(1)

        print(f"Running {args.num_gpus}-GPU inference (seed={args.seed}, steps={args.steps})...")
        ok, err = run_inference(args.cli, args.model, args.num_gpus, args.steps, args.seed, multi_out)
        if not ok:
            print(f"FAIL: Multi-GPU inference failed: {err}", file=sys.stderr)
            sys.exit(1)

        print("Comparing outputs...")
        identical, deviation, diff_bytes, total = compare_outputs(single_out, multi_out)

        if identical:
            print("PASS: Outputs are byte-identical")
            sys.exit(0)

        print(f"Deviation: {deviation:.6f}% ({diff_bytes}/{total} bytes differ)")
        if deviation < args.threshold:
            print(f"PASS: Deviation {deviation:.6f}% < threshold {args.threshold}%")
            sys.exit(0)
        else:
            print(f"FAIL: Deviation {deviation:.6f}% >= threshold {args.threshold}%")
            sys.exit(1)

if __name__ == "__main__":
    main()
```
