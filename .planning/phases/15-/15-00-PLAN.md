import pytest
import subprocess
import os

def _detect_gpu_count():
    """Detect CUDA GPU count via nvidia-smi."""
    try:
        result = subprocess.run(
            ["nvidia-smi", "-L"],
            capture_output=True, text=True, timeout=5
        )
        if result.returncode == 0:
            return len([l for l in result.stdout.strip().split("\n") if l.startswith("GPU ")])
    except (FileNotFoundError, subprocess.TimeoutExpired):
        pass
    return 0

GPU_COUNT = _detect_gpu_count()

"""Numerical precision validation tests (Plan 15-04, MGPU-09)."""
import pytest
import subprocess
import os
import struct

@pytest.mark.xfail(reason="Plan 15-04 not yet implemented")
def test_precision_script_exists():
    """Precision validation script exists."""
    script = os.path.join(os.path.dirname(__file__), "..", "scripts", "validate_precision.py")
    assert os.path.isfile(script), f"Precision script not found: {script}"

@pytest.mark.xfail(reason="Plan 15-04 not yet implemented")
def test_precision_comparison_under_threshold(require_multi_gpu, wan_cli):
    """Multi-GPU output deviates < 0.01% from single-GPU baseline.

    This test requires:
    1. A model file (set WAN_TEST_MODEL env var)
    2. At least 2 GPUs
    3. wan-cli built with CUDA

    It runs single-GPU and multi-GPU inference with identical seeds,
    then compares raw output bytes for pixel-level deviation.
    """
    model = os.environ.get("WAN_TEST_MODEL")
    if not model:
        pytest.skip("WAN_TEST_MODEL not set")

    import tempfile
    with tempfile.TemporaryDirectory() as tmpdir:
        single_out = os.path.join(tmpdir, "single.avi")
        multi_out = os.path.join(tmpdir, "multi.avi")

        # Single GPU baseline
        subprocess.run([
            wan_cli, "-m", model, "-p", "a cat", "-o", single_out,
            "--steps", "3", "--seed", "42"
        ], check=True, timeout=300)

        # Multi GPU
        subprocess.run([
            wan_cli, "-m", model, "-p", "a cat", "-o", multi_out,
            "--steps", "3", "--seed", "42", "--num-gpus", "2"
        ], check=True, timeout=300)

        # Compare file sizes first
        s1 = os.path.getsize(single_out)
        s2 = os.path.getsize(multi_out)
        assert s1 == s2, f"Output sizes differ: {s1} vs {s2}"

        # Byte-level comparison
        with open(single_out, "rb") as f1, open(multi_out, "rb") as f2:
            d1 = f1.read()
            d2 = f2.read()

        if d1 == d2:
            return  # Identical

        # Calculate deviation for non-identical outputs
        total = len(d1)
        diff_bytes = sum(1 for a, b in zip(d1, d2) if a != b)
        deviation_pct = (diff_bytes / total) * 100
        assert deviation_pct < 0.01, f"Deviation {deviation_pct:.4f}% exceeds 0.01% threshold"
